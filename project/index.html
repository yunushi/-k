import tkinter as tk def toggle_light(): global is_on is_on = not is_on if is_on: canvas.itemconfig(bulb, fill="yellow") root.config(bg="yellow") button.config(text="İpi Tekrar Çek") else: canvas.itemconfig(bulb, fill="gray") root.config(bg="black") button.config(text="İpi Çek") # Pencere oluşturma root = tk.Tk() root.title("Ampul Kontrolü") root.geometry("400x400") root.configure(bg="black") # Tuval (Canvas) oluşturma canvas = tk.Canvas(root, width=400, height=300, bg="black", highlightthickness=0) canvas.pack() # İp ve ampul canvas.create_line(200, 50, 200, 200, fill="white", width=4) # İp bulb = canvas.create_oval(150, 200, 250, 300, fill="gray", outline="white") # Ampul # Buton is_on = False button = tk.Button(root, text="İpi Çek", command=toggle_light, bg="gray", fg="white", font=("Arial", 14)) button.pack(pady=20) # Pencereyi çalıştır root.mainloop()
